class Solution {
    public int[] maximumMEX(int[] nums) {
        int n = nums.length;
        List<Integer> result = new ArrayList<>();
        
        int i = 0;
        while (i < n) {
            int mex = 0;
            boolean[] seen = new boolean[n + 2]; // Since MEX can be up to n+1 max
            
            int maxK = 0;
            int bestMex = -1;
            
            for (int k = 1; k <= n - i; k++) {
                int val = nums[i + k - 1];
                if (val < seen.length) {
                    seen[val] = true;
                }
                
                // Compute current MEX
                while (seen[mex]) {
                    mex++;
                }
                
                // Track the best (largest) MEX found so far
                if (mex > bestMex) {
                    bestMex = mex;
                    maxK = k;
                }
            }
            
            // Append the best MEX and jump by maxK
            result.add(bestMex);
            i += maxK;
        }
        
        // Convert List to array
        int[] ans = new int[result.size()];
        for (int j = 0; j < result.size(); j++) {
            ans[j] = result.get(j);
        }
        return ans;
    }
}
