class Solution {
    public int[] lexicographicallyMaximumMEXArray(int[] nums) {
        int n = nums.length;
        List<Integer> result = new ArrayList<>();
        List<Integer> current = new ArrayList<>();
        for (int num : nums) current.add(num);
        
        while (!current.isEmpty()) {
            int bestMex = -1;
            int bestK = -1;
            
            for (int k = 1; k <= current.size(); k++) {
                int mex = findMEX(current, k);
                if (mex > bestMex) {
                    bestMex = mex;
                    bestK = k;
                }
            }
            
            result.add(bestMex);
            // Remove first bestK elements
            current = current.subList(bestK, current.size());
        }
        
        int[] ans = new int[result.size()];
        for (int i = 0; i < ans.length; i++) {
            ans[i] = result.get(i);
        }
        return ans;
    }
    
    private int findMEX(List<Integer> arr, int k) {
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < k; i++) {
            set.add(arr.get(i));
        }
        int mex = 0;
        while (set.contains(mex)) mex++;
        return mex;
    }
}
