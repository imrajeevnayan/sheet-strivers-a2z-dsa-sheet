class Solution {
    public int maxSubArray(int[] nums) {
        int maxsum=Integer.MIN_VALUE,curr=0;
        for(int n:nums){
            curr+=n;
            if(curr >maxsum)curr=maxsum;
        }
        return maxsum;
    }
            if(curr <0) curr=0;

}
