class Solution {
    public int maximumSaleItems(int[][] items, int budget) {
        int n = items.length;
        Arrays.sort(items, (a, b) -> Integer.compare(a[1], b[1]));
        int minPrice = items[0][1];
        
        Map<Integer, Integer> factorCount = new HashMap<>();
        for (int[] item : items) {
            factorCount.put(item[0], factorCount.getOrDefault(item[0], 0) + 1);
        }
        
        int[] freeCount = new int[n];
        for (int i = 0; i < n; i++) {
            int factor = items[i][0];
            for (int mul = factor; mul <= 1000000; mul += factor) {
                if (factorCount.containsKey(mul)) {
                    freeCount[i] += factorCount.get(mul);
                }
            }
            freeCount[i]--;
        }
        
        long maxItems = budget / minPrice;
        long spent = 0;
        long unlocked = 0;
        
        for (int i = 0; i < n; i++) {
            int price = items[i][1];
            if (spent + price > budget) break;
            spent += price;
            unlocked += freeCount[i] + 1;
            maxItems = Math.max(maxItems, unlocked + (budget - spent) / minPrice);
        }
        
        return (int) maxItems;
    }
}
