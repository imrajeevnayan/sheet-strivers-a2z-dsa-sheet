class Solution {
    public int findNthDigit(int n) {
        long len = 1, count = 9, start = 1;

        // 1. Find which "digit-length" group n falls into
        while (n > len * count) {
            n -= len * count;
            len++;
            count *= 10;
            start *= 10;
        }

        // 2. Find the specific number
        start += (n - 1) / len;

        // 3. Return the specific digit from that number
        return String.valueOf(start).charAt((int)((n - 1) % len)) - '0';
    }
}
