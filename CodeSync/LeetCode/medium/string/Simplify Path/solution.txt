for (String part : parts) {
            if (part.equals("..")) {
                if (!stack.isEmpty()) stack.removeLast();
            } else if (!part.equals("") && !part.equals(".")) {
                stack.addLast(part);
            }
        }

        if (stack.isEmpty()) return "/";

        StringBuilder sb = new StringBuilder();
        for (String dir : stack) {
            sb.append("/").append(dir);
        }
        return sb.toString();
    }
