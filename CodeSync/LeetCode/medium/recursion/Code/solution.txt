            while (!stack.isEmpty() && stack.peek().val < curr.val) {
                stack.pop();
            }
            stack.push(curr);
            curr = curr.next;
        }
        
        // Rebuild result list from stack (top of stack = last surviving node)
        ListNode next = null;
        while (!stack.isEmpty()) {
            ListNode node = stack.pop();
            node.next = next;
            next = node;
        }
        
        return next;
    }
