        private int robLinear(int[] nums, int start, int end) {
        int prev1 = 0; // Max profit up to i-1
        int prev2 = 0; // Max profit up to i-2
        
        for (int i = start; i <= end; i++) {
            int current = Math.max(prev1, prev2 + nums[i]);
            prev2 = prev1;
            prev1 = current;
        }
        return prev1;
    }
