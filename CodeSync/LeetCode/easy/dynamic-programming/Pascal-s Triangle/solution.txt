public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> triangle = new ArrayList<>();

        for (int i = 0; i < numRows; i++) {
            List<Integer> row = new ArrayList<>();
            long ans = 1; // Use long to prevent overflow during calculation

            for (int j = 0; j <= i; j++) {
                // Add the current calculated value
                row.add((int) ans);
                
                // Calculate next value in the same row
                // Formula: C(i, j+1) = C(i, j) * (i - j) / (j + 1)
                ans = ans * (i - j) / (j + 1);
            }
            
            triangle.add(row);
        }
        return triangle;
    }
