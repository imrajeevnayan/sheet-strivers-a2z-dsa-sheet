class Solution {
    public ListNode reverseList(ListNode head) {
        ListNode prev=null;
        ListNode curr=head;
        while(curr!=null){
            ListNode temp=curr.next;
        }
        return prev;
    }
            curr.next=prev;
            prev=curr;
            curr=temp;
}
