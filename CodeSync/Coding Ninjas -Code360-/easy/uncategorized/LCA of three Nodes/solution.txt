public class Solution {
    public static Node reverseLL(Node head, int low, int high) {
        if (head == null || low == high) return head;
        
        Node dummy = new Node(0);
        dummy.next = head;
        Node prev = dummy;
        
        for (int i = 1; i < low; i++) {
            prev = prev.next;
        }
        
        Node curr = prev.next;
        Node next = null;
        Node prevRev = null;
        Node tail = curr;
        
        for (int i = low; i <= high && curr != null; i++) {
            next = curr.next;
            curr.next = prevRev;
            prevRev = curr;
            curr = next;
        }
        
        tail.next = curr;
        prev.next = prevRev;
        
        return dummy.next;
    }
