public class Solution {
 
    public static Node<Integer> mergeSort(Node<Integer> head) {
        if (head == null || head.next == null) {
            return head;
        }
 
        Node<Integer> mid = getMid(head);
        Node<Integer> secondHalf = mid.next;
        mid.next = null; // Break the link
 
        Node<Integer> left = mergeSort(head);
        Node<Integer> right = mergeSort(secondHalf);
 
        return merge(left, right);
    }
 
    private static Node<Integer> getMid(Node<Integer> head) {
        if (head == null) return head;
        Node<Integer> slow = head;
        Node<Integer> fast = head.next;
        
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
        }
        return slow;
    }
