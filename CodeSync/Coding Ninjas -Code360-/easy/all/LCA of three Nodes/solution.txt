public class Solution {
    public static Node<Integer> mergeSort(Node<Integer> head) {
        if (head == null || head.next == null) {
            return head;
        }
        
        Node<Integer> mid = findMiddle(head);
        Node<Integer> rightHead = mid.next;
        mid.next = null; // Split the list into two halves
        
        Node<Integer> leftSorted = mergeSort(head);
        Node<Integer> rightSorted = mergeSort(rightHead);
        
        return merge(leftSorted, rightSorted);
    }
    
    private static Node<Integer> findMiddle(Node<Integer> head) {
        if (head == null) {
            return head;
        }

        // When fast reaches the end, slow will be at the middle
