        PriorityQueue<int[]> pq = new PriorityQueue<>((x, y) -> x[0] - y[0]);
        
        int currentMax = Integer.MIN_VALUE;
        
        // Initialize heap with first element from each list
        for (int i = 0; i < k; i++) {
            pq.offer(new int[]{a[i][0], i, 0});
            currentMax = Math.max(currentMax, a[i][0]);
        }
