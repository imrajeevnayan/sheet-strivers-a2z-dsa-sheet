import java.util.*;
import java.io.*;
 
public class FrequencyStack {
    private Map<Integer, Integer> freqMap;
    private Map<Integer, Deque<Integer>> groupMap;
    private int maxFreq;
 
    FrequencyStack() {
        freqMap = new HashMap<>();
        groupMap = new HashMap<>();
        maxFreq = 0;
    }
 
    void push(int element) {
        int freq = freqMap.getOrDefault(element, 0) + 1;
        freqMap.put(element, freq);
        maxFreq = Math.max(maxFreq, freq);
 
        groupMap.computeIfAbsent(freq, k -> new ArrayDeque<>()).push(element);
    }
 
    int pop() {
        Deque<Integer> stack = groupMap.get(maxFreq);
        int element = stack.pop();
 
        // Update frequency
        freqMap.put(element, freqMap.get(element) - 1);
