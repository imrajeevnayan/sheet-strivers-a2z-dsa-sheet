import java.util.*;
import java.io.*;
 
public class Solution {
    public static int[] findSubarray(int[] arr) {
        Map<Integer, Integer> prefixMap = new HashMap<>();
        prefixMap.put(0, -1); // Handle subarrays starting from index 0
        
        int maxLen = 0;
        int startIdx = -1;
        int prefixSum = 0;
        
        for (int i = 0; i < arr.length; i++) {
            prefixSum += arr[i];
            
            if (prefixMap.containsKey(prefixSum)) {
                int len = i - prefixMap.get(prefixSum);
                if (len > maxLen) {
                    maxLen = len;
                    startIdx = prefixMap.get(prefixSum) + 1;
                }
            } else {
                prefixMap.put(prefixSum, i);
            }
        }
        
        return new int[]{startIdx, startIdx + maxLen - 1};
    }
}
