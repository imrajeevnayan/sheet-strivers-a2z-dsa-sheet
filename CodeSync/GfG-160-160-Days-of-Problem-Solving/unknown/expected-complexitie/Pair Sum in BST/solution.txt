class Solution {
    boolean findTarget(Node root, int target) {
        HashSet<Integer> set = new HashSet<>();
        return findTargetHelper(root, target, set);
    }
    
    private boolean findTargetHelper(Node root, int target, HashSet<Integer> set) {
        if (root == null) return false;
        
         if (set.contains(target - root.data)) return true;
           set.add(root.data);
        
        // Check left and right subtrees
        return findTargetHelper(root.left, target, set) || 
               findTargetHelper(root.right, target, set);
    }
}
