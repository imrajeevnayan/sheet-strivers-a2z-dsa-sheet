class Solution {
    public Node rotate(Node head, int k) {
        if (head == null || k == 0) return head;
        
        Node slow = head;
        Node fast = head;
        
        for (int i = 0; i < k; i++) {
            if (fast == null) return head;
            fast = fast.next;
        }
        
        if (fast == null) return head;
        
        while (fast.next != null) {
            slow = slow.next;
            fast = fast.next;
        }
        
        Node newHead = slow.next;
        slow.next = null;
        fast.next = head;
        
        return newHead;
    }
}
