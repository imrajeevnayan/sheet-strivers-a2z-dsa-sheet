class Solution {
    public Node rotate(Node head, int k) {
        if (head == null || k == 0) return head;
        
        Node curr = head;
        int count = 1;
        
        while (curr.next != null) {
            curr = curr.next;
            count++;
        }
        
        k = k % count;
        if (k == 0) return head;
        
        curr.next = head;
        
        Node newTail = head;
        for (int i = 1; i < k; i++) {
            newTail = newTail.next;
        }
        
        Node newHead = newTail.next;
        newTail.next = null;
        
        return newHead;
    }
}
