class Solution {
    public Node rotate(Node head, int k) {
        if (head == null || head.next == null || k == 0) return head;
        Node temp = head;
        int len = 1;

        while (temp.next != null) {
            temp = temp.next;
            len++;
        }

        // Step 2: reduce k
        k = k % len;
        if (k == 0) return head;

        // Step 3: go to k-th node
        Node curr = head;
        for (int i = 1; i < k; i++) {
            curr = curr.next;
        }

        Node newHead = curr.next;
        curr.next = null;

        // Step 4: connect last to head
        temp.next = head;

        return newHead;
    }
}
