class Solution {
    public static int findEquilibrium(int arr[]) {
        int n = arr.length;
        long totalSum = 0;
        
        for (int num : arr)totalSum += num;
        long leftSum = 0;
        for (int i = 0; i < n; i++) {
            long rightSum = totalSum - leftSum - arr[i];
            if (leftSum == rightSum)  return i;
            leftSum += arr[i];
        }
        return -1;
    }
}
