class Solution {
    boolean findTarget(Node root, int target) {
        if (root == null) return false;
        if (root.data == target) return true;
        
        
        if (target < root.data) {
            return findTarget(root.left, target);
        } else {
            return findTarget(root.right, target);
        }
    }
}
