import java.util.*;

// Without DRY (duplicated logic)
public class Main {
    public static boolean validateEmailFormat1(String email) {
        return email.contains("@") && email.contains(".");
    }
​
    public static boolean validateEmailFormat2(String email) {
        return email.contains("@") && email.contains(".");
    }
​
    // With DRY (single reusable function)
    public static boolean validateEmailFormat(String email) {
        return email.contains("@") && email.contains(".");
    }
​
    public static void main(String[] args) {
        String email1 = "user@example.com";
        String email2 = "invalidemail";
​
        // Without DRY usage
        System.out.println("Without DRY validation 1: " + validateEmailFormat1(email1));
        System.out.println("Without DRY validation 2: " + validateEmailFormat2(email2));
​
        // With DRY usage
        System.out.println("With DRY validation: " + validateEmailFormat(email1));
    }
}
