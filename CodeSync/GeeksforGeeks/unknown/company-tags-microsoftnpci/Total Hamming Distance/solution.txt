class Solution {
    int totHammingDist(int arr[]) {
        int n = arr.length;
        int total = 0;

        for (int b = 0; b < 32; b++) {
            int countOnes = 0;
            for (int num : arr) {
                if ((num & (1 << b)) != 0) {
                    countOnes++;
                }
            }
            total += countOnes * (n - countOnes);
        }

        return total;
    }
}
