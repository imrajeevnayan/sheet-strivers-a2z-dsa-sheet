class Solution {
    public static ArrayList<ArrayList<Integer>> evenOdd(ArrayList<Integer> list) {
        ArrayList<Integer> even = new ArrayList<>();
        ArrayList<Integer> odd = new ArrayList<>();

        for (int num : list) {
            if (num % 2 == 0)
                even.add(num);
            else
                odd.add(num);
        }

        ArrayList<ArrayList<Integer>> result = new ArrayList<>();
        result.add(even);
        result.add(odd);

        return result;
    }
}
