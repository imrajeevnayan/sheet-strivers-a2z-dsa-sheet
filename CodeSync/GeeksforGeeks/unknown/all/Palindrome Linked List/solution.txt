public boolean isPalindrome(Node head) {
    Stack<Integer> st = new Stack<>();
    Node curr = head;

    // 1. Push all elements into the stack
    while (curr != null) {
        st.push(curr.data);
        curr = curr.next;
    }

    // 2. Compare elements from start with popped elements from stack
    curr = head;
    while (curr != null) {
        if (curr.data != st.pop()) {
            return false;
        }
        curr = curr.next;
    }

    return true;
}
