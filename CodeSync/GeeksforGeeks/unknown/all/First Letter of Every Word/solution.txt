class Solution {
    String firstAlphabet(String s) {
        String[] words = s.split(" ");
        StringBuilder res = new StringBuilder();

        for (String w : words) {
            if (w.length() > 0) {
                res.append(w.charAt(0));
            }
        }
        return res.toString();
    }
}
