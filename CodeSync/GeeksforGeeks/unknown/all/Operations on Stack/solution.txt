class Solution {

    void insert(Stack<Integer> s, int x) {
        s.push(x);
    }

    void remove(Stack<Integer> s) {
        if (!s.isEmpty()) {
            s.pop();
        }
    }

    void headOf_Stack(Stack<Integer> s) {
        if (!s.isEmpty()) {
            System.out.println(s.peek());
        }
    }

    boolean find(Stack<Integer> s, int val) {
        return s.contains(val);
    }
}
