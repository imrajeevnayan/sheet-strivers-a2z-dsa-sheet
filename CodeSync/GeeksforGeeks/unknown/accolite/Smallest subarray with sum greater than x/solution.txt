class Solution {
    public static int smallestSubWithSum(int x, int[] arr) {
        int n = arr.length;
        int minLen = Integer.MAX_VALUE;
        int sum = 0,start = 0;
        
        for(int end = 0; end < n; end++) {
            sum += arr[end];
            while(sum > x) {
                minLen = Math.min(minLen, end - start + 1);
                sum -= arr[start++];
            }
        }
        return (minLen == Integer.MAX_VALUE) ? 0 : minLen;
    }
}
