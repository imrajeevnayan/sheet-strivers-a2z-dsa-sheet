class Solution {
    public boolean hasTripletSum(int arr[], int target) {
     int n = arr.length;
    for (int i = 0; i < n - 2; i++) {
        HashSet<Integer> set = new HashSet<>();
        for (int j = i + 1; j < n; j++) {
            int sum = target - arr[i] - arr[j];
            if (set.contains(sum))return true;
            set.add(arr[j]);
            }
         }
       return false;
    }
}
