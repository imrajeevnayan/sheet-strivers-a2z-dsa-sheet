import java.util.Stack;

class Solution {
    public Node reverseCircular(Node head) {
        if (head == null || head.next == head) return head;

        Stack<Node> stack = new Stack<>();
        Node curr = head;

        // traverse circular list using while
        while (true) {
            stack.push(curr);
            curr = curr.next;
            if (curr == head) break;
        }

        // new head after reversal
        Node newHead = stack.pop();
        Node temp = newHead;

        // rebuild reversed links
        while (!stack.isEmpty()) {
            temp.next = stack.pop();
            temp = temp.next;
        }

        // make circular again
        temp.next = newHead;

        return newHead;
    }
}
