class Solution {
    private Node prev = null;

    public Node flatten(Node root) {
        if (root == null) return null;

        // Traverse right subtree first, then left
        flatten(root.right);
        flatten(root.left);

        // Set the current node's right to prev and left to null
        root.right = prev;
        root.left = null;
        
        // Update prev to current node
        prev = root;

        return root;
    }
}
