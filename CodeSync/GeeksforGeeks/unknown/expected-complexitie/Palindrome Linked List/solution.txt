class Solution{
public boolean isPalindrome(Node head) {
    Stack<Integer> st = new Stack<>();
    Node curr = head;
    while (curr != null) {
        st.push(curr.data);
        curr = curr.next;
    }

curr = head;
    while (curr != null) {
        if (curr.data != st.pop()) {
            return false;
        }
        curr = curr.next;
    }

    return true;
   }
}
