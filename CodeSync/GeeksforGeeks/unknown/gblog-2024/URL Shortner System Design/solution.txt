std::string to_base_62(int deci) {
    std::string s = "012345689abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ";
    std::string hash_str = "";
    while (deci > 0) {
        hash_str = s[deci % 62] + hash_str;
        deci /= 62;
    }
    return hash_str;
}
